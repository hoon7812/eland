$(window).ready(function() {
	var fs_opt = { pagination: '.list_roll_fs .d_indi' , speed: 300 , autoplay: 3000 , simulateTouch : false , createPagination: false , paginationElement : "a" , paginationClickable : true , loop : true}
	if($('.box_fs .swiper-slide').size() <= 1) {
		$('.box_fs .set_indicator').hide();
		$('.box_fs .fs_controll').hide();
		return;
	}
	var fsBnr = new Swiper('.list_roll_fs .swiper-container',fs_opt);

	var swiperEle = {
		$target : $('.list_roll_fs')
		,status : true
		,last_idx : 0
		,ins : fsBnr
	};// 최상단 박스 , $target : autoplay - play , status : stop 버튼 관련 , last_idx : autoplay 관련 , ins : swiper instance name

	swiperEle.$target.hover(
		function (e) { //mouseenter
			swiperEle.ins.stopAutoplay();
			e.preventDefault();
		},
		function (e) { //mouseleave
			if ( swiperEle.status ) swiperEle.ins.startAutoplay();
			e.preventDefault();
		}
	);

	swiperEle.$target.find('.d_indi a').on('click',function(e){
		e.preventDefault();
	});

	swiperEle.$target.find('.d_play').on('click', function(e){
		swiperEle.status = true;
		swiperEle.ins.startAutoplay();

		$(this).parent().removeClass("ons");
		swiperEle.$target.find('.d_stop').focus();
		e.preventDefault();
	});

	swiperEle.$target.find('.d_stop').on('click', function(e){
		swiperEle.status = false;
		swiperEle.ins.stopAutoplay();

		$(this).parent().addClass("ons");
		swiperEle.$target.find('.d_play').focus();
		e.preventDefault();
	});

	swiperEle.$target.on("keydown", ".swiper-slide" , function(e) {
		var idx = Math.floor($(this).attr('data-index'));
		if(e.keyCode == 9){ swiperEle.ins.swipeTo(idx, 0 , 0 );}
		swiperEle.last_idx = idx;
	});

	var end_idx = parseInt(swiperEle.$target.find(".swiper-slide").data("index"))+1; // 마지막 blur 처리
	swiperEle.$target.find(".swiper-slide").eq(end_idx).on("blur", "a:last" ,function(e){
		swiperEle.ins.startAutoplay();
		e.preventDefault();
	})

	swiperEle.$target.on("focusin" , "a" , function(e) {
		swiperEle.ins.stopAutoplay();
	});

	swiperEle.$target.on("focus", ".swiper-slide" ,function(){
		var idx = Math.floor($(this).attr('data-index'));
		swiperEle.ins.stopAutoplay();
		swiperEle.ins.swipeTo( idx, 0 , 0 )
	});

	swiperEle.$target.find('.d_prev_fs').on('click',function(e) {
		swiperEle.ins.swipePrev();
		e.preventDefault();
	});
	swiperEle.$target.find('.d_next_fs').on('click',function(e) {
		swiperEle.ins.swipeNext();
		e.preventDefault();
	});

	swiperEle.$target.find(".swiper-slide-duplicate a").attr('tabindex','-1');
	swiperEle.$target.find(".swiper-slide:not(.swiper-slide-duplicate)").attr('tabindex','0');

});
